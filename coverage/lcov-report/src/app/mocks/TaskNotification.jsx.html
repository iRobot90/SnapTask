import { useState, useEffect } from 'react';
import { Bell, CheckCircle, X } from 'lucide-react';
import { PageConfig } from '@fleetbo';
 
export default function TaskNotification() {
  const [visible, setVisible] = useState(false);
  const [closing, setClosing] = useState(false);
 
  // Simulate receiving the payload from the JS trigger
  const notificationData = {
    title: "New Mission Assigned",
    body: "Task #402: Server Migration",
    time: "Now"
  };
 
  useEffect(() => {
    // 1. Play System Sound
    const audio = new Audio('https://fleetbo.io/audio/console/notification.mp3');
    audio.play().catch(() => {});
 
    // 2. Show Notification Banner
    setTimeout(() => setVisible(true), 100);
 
    // 3. Auto-close simulation after 4 seconds
    const timer = setTimeout(() => {
      handleClose();
    }, 4000);
 
    return () => clearTimeout(timer);
  }, []);
 
  const handleClose = () => {
    setClosing(true);
    setTimeout(() => {
      // Return success to the pilot
      if (window.Fleetbo) {
        window.Fleetbo.close(true);
      }
    }, 300);
  };
 
  return (
    <div 
      className="d-flex flex-column align-items-center justify-content-start"
      style={{ 
        width: '100%', 
        height: '100vh', 
        backgroundColor: 'rgba(0,0,0,0.2)', // Dimmed background
        backdropFilter: 'blur(4px)',
        paddingTop: '20px',
        overflow: 'hidden'
      }}
    >
      <PageConfig navbar="none" />
 
      {/* Notification Banner Simulation */}
      <div 
        className="d-flex align-items-start p-3 rounded-3 shadow-lg"
        style={{ 
          width: '90%',
          maxWidth: '350px',
          backgroundColor: '#1e1e1e',
          border: '1px solid #333',
          color: '#fff',
          transform: visible && !closing ? 'translateY(0)' : 'translateY(-150%)',
          transition: 'transform 0.4s cubic-bezier(0.175, 0.885, 0.32, 1.275)',
          cursor: 'default'
        }}
        onClick={handleClose}
      >
        {/* Icon Section */}
        <div 
          className="rounded-circle d-flex align-items-center justify-content-center me-3"
          style={{ 
            width: 40, 
            height: 40, 
            backgroundColor: '#2563eb', // Fleetbo Blue
            flexShrink: 0
          }}
        >
          <Bell size={20} color="#fff" />
        </div>
 
        {/* Content Section */}
        <div className="flex-grow-1">
          <div className="d-flex justify-content-between align-items-center mb-1">
            <span style={{ fontSize: '0.85rem', fontWeight: 600, color: '#fff' }}>
              {notificationData.title}
            </span>
            <span style={{ fontSize: '0.7rem', color: '#888' }}>
              {notificationData.time}
            </span>
          </div>
          <div style={{ fontSize: '0.8rem', color: '#ccc', lineHeight: '1.3' }}>
            {notificationData.body}
          </div>
        </div>
      </div>
 
      {/* Visual Feedback for Developer */}
      <div 
        className="position-absolute bottom-0 w-100 p-4 text-center"
        style={{ color: '#fff', opacity: 0.6, fontSize: '0.8rem' }}
      >
        <CheckCircle size={24} className="mb-2" />
        <p>System Notification Channel Active</p>
        <div 
            className="d-inline-flex align-items-center justify-content-center mt-2"
            style={{ 
                width: 50, height: 50, 
                borderRadius: '50%', 
                backgroundColor: 'rgba(255,255,255,0.1)',
                cursor: 'default'
            }}
            onClick={handleClose}
        >
            <X size={20} />
        </div>
      </div>
    </div>
  );
}
