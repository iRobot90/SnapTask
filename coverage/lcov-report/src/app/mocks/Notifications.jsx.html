import { useState, useEffect } from 'react';
import { PageConfig } from '@fleetbo';
import { Bell, Send, X } from 'lucide-react';
 
export default function Notifications() {
  const [activeNotification, setActiveNotification] = useState(null);
  const [history, setHistory] = useState([]);
 
  // Simulate receiving a notification command
  const triggerNotification = () => {
    const newNotif = {
      id: Date.now(),
      title: 'System Alert',
      body: 'Upload complete. 120FPS achieved.',
      time: new Date().toLocaleTimeString()
    };
 
    // 1. Visual Feedback (Banner)
    setActiveNotification(newNotif);
    
    // 2. Audio Feedback (Fleetbo Bank)
    const audio = new Audio('https://fleetbo.io/audio/console/notification.mp3');
    audio.play().catch(() => {}); // Ignore autoplay restrictions
 
    // 3. Add to history log
    setHistory(prev => [newNotif, ...prev]);
 
    // Auto-hide banner after 3s
    setTimeout(() => setActiveNotification(null), 3000);
  };
 
  const closeModule = () => {
    // Return the history or just close
    // @ts-ignore
    if (window.Fleetbo) window.Fleetbo.close({ sentCount: history.length });
  };
 
  return (
    <div 
      className="d-flex flex-column bg-light"
      style={{ 
        width: '100%', 
        height: '100vh', 
        overflow: 'hidden',
        position: 'relative'
      }}
    >
      <PageConfig navbar="none" />
 
      {/* --- HEADER --- */}
      <div className="d-flex align-items-center justify-content-between px-4 py-3 bg-white shadow-sm">
        <div className="d-flex align-items-center">
          <Bell size={20} className="text-primary me-2" />
          <span className="fw-bold text-dark">Notification Center</span>
        </div>
        <div 
          onClick={closeModule}
          style={{ cursor: 'default' }}
          className="p-2"
        >
          <X size={24} className="text-secondary" />
        </div>
      </div>
 
      {/* --- MAIN CONTENT --- */}
      <div className="flex-grow-1 p-4 d-flex flex-column align-items-center justify-content-center">
        <div 
          className="bg-white rounded-circle d-flex align-items-center justify-content-center shadow-sm mb-4"
          style={{ width: 120, height: 120 }}
        >
          <Bell size={48} className="text-primary" />
        </div>
        
        <h5 className="text-dark fw-bold mb-2">Test Channel</h5>
        <p className="text-muted text-center mb-4" style={{ maxWidth: 280 }}>
          Trigger a simulated native notification to test the audio and visual overlay.
        </p>
 
        <div
          className="bg-dark text-white rounded-pill d-flex align-items-center justify-content-center px-4 py-3 shadow"
          style={{ 
            cursor: 'default',
            transition: 'transform 0.1s',
            minWidth: 200
          }}
          onTouchStart={(e) => e.currentTarget.style.transform = 'scale(0.95)'}
          onTouchEnd={(e) => e.currentTarget.style.transform = 'scale(1)'}
          onMouseDown={(e) => e.currentTarget.style.transform = 'scale(0.95)'}
          onMouseUp={(e) => e.currentTarget.style.transform = 'scale(1)'}
          onClick={triggerNotification}
        >
          <Send size={18} className="me-2" />
          <span className="fw-bold">Send Test Alert</span>
        </div>
      </div>
 
      {/* --- SIMULATED NATIVE BANNER --- */}
      {activeNotification && (
        <div 
          className="position-absolute top-0 start-0 w-100 p-3"
          style={{ zIndex: 9999, animation: 'slideDown 0.3s ease-out' }}
        >
          <div className="bg-white rounded shadow-lg p-3 d-flex align-items-start border-start border-5 border-primary">
            <div className="me-3">
              <Bell size={20} className="text-primary" />
            </div>
            <div>
              <div className="d-flex justify-content-between align-items-center mb-1">
                <span className="fw-bold text-dark small">{activeNotification.title}</span>
                <span className="text-muted ms-3" style={{ fontSize: 10 }}>now</span>
              </div>
              <div className="text-secondary small" style={{ lineHeight: 1.2 }}>
                {activeNotification.body}
              </div>
            </div>
          </div>
        </div>
      )}
 
      {/* --- HISTORY LIST --- */}
      <div className="bg-white border-top p-3" style={{ height: '30%' }}>
        <h6 className="text-muted small fw-bold mb-3">RECENT LOGS</h6>
        <div className="overflow-auto" style={{ height: 'calc(100% - 30px)' }}>
          {history.length === 0 ? (
            <div className="text-muted small fst-italic">No notifications sent yet.</div>
          ) : (
            history.map((item) => (
              <div key={item.id} className="d-flex align-items-center mb-2 pb-2 border-bottom">
                <div className="bg-light rounded p-1 me-2">
                  <Bell size={12} className="text-secondary" />
                </div>
                <div className="flex-grow-1">
                  <div className="text-dark small fw-bold">{item.title}</div>
                  <div className="text-muted" style={{ fontSize: 10 }}>{item.time}</div>
                </div>
              </div>
            ))
          )}
        </div>
      </div>
 
      <style>{`
        @keyframes slideDown {
          from { transform: translateY(-100%); opacity: 0; }
          to { transform: translateY(0); opacity: 1; }
        }
      `}</style>
    </div>
  );
}
